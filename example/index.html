如何使用？


popup({
    title : '这是标题',
    content : '这是内容',
    layer_close : false,
    confirm : function(){
        console.log('click confirm');
    },
    concel : function(){
        console.log('click concel');
    }
});

如果你需要修改确认按钮和取消按钮的文字内容：


popup({
    title : '这是标题',
    content : '这是内容',
    layer_close : false,
    confirm : {
        text:"保存"   // 默认值是“确认”,
        callback:function(){
            console.log('click confirm');
        },
    }
    concel : {
        text:"否"   // 默认值是“取消”,
        callback:function(){
            console.log('click concel');
        },
    }
});
